    \frac{d}{dt}C_{D}(t) & = r_{1} - r_{2}
    \\

    \frac{d}{dt}C_{G}(t) & = r_{3}
    \\
    \frac{d}{dt}C_{E}(t) & = r_{1} + r_{2} + r_{3}
